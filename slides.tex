      \onslide<5->{
        \begin{itemize}

          \item<7-> \hlmain{Probability (for capturing phenomena with uncertainty).}

        \end{itemize}
      }
    \item<+-> \hlattn{Building models using mathematical objects.}
      \onslide<6->{

          \item shapes and change.
        \end{itemize}
      }

    \item<+-> How to be a scientist in 2024 and beyond.
      \onslide<9->{
        \begin{itemize}

      }
  \end{itemize}
