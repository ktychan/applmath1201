\begin{exercise}
  Find the general solution for \(\frac{du}{dt} + u(t) = t^{2}e^{-t}\).

  \begin{solution}
    This is a linear, first-order DE. We need to start by finding the integrating factor
    \[
      v(t) = e^{\int 1 \;dx} = e^{t}.
    \]

    If we remember the formula, we can directly use it. Otherwise, we multiply through by \(v(t)\) to get
    \[
      (u(t) v(t))' = v(t) t^{2} e^{-t} = t^{2}.
    \]
    Integrate both sides to get
    \[
      u(t) = e^{-t} \int t^{2} \;dt = e^{-t}\left( \frac{t^{3}}{3} + C \right).
    \]
  \end{solution}
\end{exercise}

\begin{exercise}
  Solve the initial-value problem \(u'(t) + tu(t) = 3t\) with \(u(0) = 1\).  

  \begin{solution}
    This is a linear, first-order DE. We need to start by finding the integrating factor
    \[
      v(t) = e^{\int t \;dx} = e^{t^{2}/2}.
    \]

    If we remember the formula, we can directly use it. Personally, I prefer to multiply through by \(v(t)\), so I don't forget that \(v(t)\) is supposed to show up on the right-hand side.
    \[
      (u(t) v(t))' = v(t) 3t
    \]
    Integrate both sides to get
    \[
      u(t) = e^{-t^{2}/2} \int 3t e^{t^{2}/2} \;dt = e^{-t^{2}/2}\left( 3t^{t^{2}/2} + C\right) = 3 + Ce^{-t^{2}/2}.
    \]
    The last integration requires a substitution \(w = t^{2}/2\).


    Using \(u(0) = 1\) we get \(C = -2\). The solution is \(3 - 2 e^{-t^{2}/2}\).
  \end{solution}
\end{exercise}

\begin{exercise}
  Find the general solution for \(y'(x) + \frac{y(x)}{x^{2} + 1} = 0\).

  \begin{solution}
    We have a linear, first-order DE. Find the integrating factor
    \[
      v(t) = e^{\int \frac{1}{x^{2} + 1}} = e^{\arctan(x)}.
    \]
    
    It is quite fast to use the formula here, but remember that \(\int 0 \;dt = C\).
    \[
      u(t) = \frac{1}{v(t)} \int 0 \;dt = Ce^{-\arctan(x)}.
    \]
  \end{solution}
\end{exercise}

\begin{exercise}
  Find \(\lim_{x \to \infty} y(x)\) given that \(y(x)\) satisfies
  \[
    \frac{dy}{dx} + y = e^{x} \text{ and \(y(0) = \frac{3}{2}\)}.
  \]

  \begin{solution}
    This is NOT an autonomous equation, so we cannot use the phase-line analysis. However, we can solve the DE.

    Find the integrating factor \(v(x) = e^{\int 1 \;dx} = e^{x}\). The formula says 
    \[
      y(x) = \frac{1}{v(t)} \int v(t) e^{x}\;dx = e^{-x} \int e^{2x} \;dx = e^{-x}\left( \frac{1}{2} e^{2x} + C \right) = \frac{e^{x}}{2} + C e^{-x}.
    \]
    The integration requires the substitution \(w = 2x\).

    Using the initial condition \(y(0) = 3/2\), we get \(C = 1\). The solution is
    \[
      y(x) = \frac{e^{x}}{2} + e^{-x}.
    \]

  
    We see that \(\lim_{x \to \infty} y(x) = \infty\).
  \end{solution}
\end{exercise}

\begin{exercise}
  Find the general solution for \(t y'(t) - 2y(t) = t^{2}\) for \(t > 0\).

  \begin{solution}
    We have a linear, first-order DE but not in standard form.  We must start by rewriting it in the standard form 
    \[
      y'(t) - \frac{2}{t} y(t) = t.
    \]

    Now we can find the integrating factor
    \[
      v(t) = e^{\int -\frac{2}{t} \;dt} = e^{-2\ln|t|} = t^{-2}.
    \]

    Using the formula, we get
    \[
      y(t) = t^{2} \int t^{-2} t \;dt = t^{2} \int \frac{1}{t} \;dt = t^{2}\left( \ln(t) + C \right).
    \]
    Since \(t > 0\), we don't need take absolute values inside \(\ln(\ldots)\).
  \end{solution}
\end{exercise}

\begin{exercise}
  Solve the initial-value problem \(t u'(t) + u(t) = \frac{\ln(t)}{t}\) and \(u(1) = 0\).

  \begin{solution}
    We have a linear, first-order DE but not in standard form. Rewrite it to get
    \[
      u'(t) + \frac{1}{t} u(t) = \frac{\ln(t)}{t^{2}}.
    \]

    The integrating factor is \(v(t) = e^{\int \frac{1}{t} \;dt} = e^{\ln|t|} = |t|\).  We need to deal with the absolute values.

    If \(t > 0\), then \(v(t) = |t| = t\) and 
    \[
      u(t) = \frac{1}{t} \int t \frac{\ln(t)}{t^{2}} \;dt = \frac{1}{t} \int \frac{\ln(t)}{t} \;dt = \frac{1}{t}\left( \frac{\left( \ln(t) \right)^{2}}{2} + C \right).
    \]

    If \(t < 0\), then \(v(t) = |t| = -t\) and
    \[
      u(t) = \frac{1}{-t} \int -t \frac{\ln(t)}{t^{2}} \;dt = \frac{1}{t} \int \frac{\ln(t)}{t} \;dt = \frac{1}{-t}\left( \frac{\left( \ln(-t) \right)^{2}}{2} + C \right).
    \]

    The two solutions can be combined into just 
    \[
      u(t) = \frac{1}{|t|}\left( \frac{\left( \ln|t| \right)^{2}}{2} + C \right), \text{ for } t \ne 0.
    \]

    Using the initial condition \(u(1) = 0\), we get \(C = 0\). After a little bit of optional tidying up, the solution is
    \[
      u(t) = \frac{\left( \ln|t| \right)^{2}}{2|t|}, \text{ for } t \ne 0.
    \]
  \end{solution}
\end{exercise}

\begin{exercise}
  Solve the initial-value problem \(y'(t) + t y(t) = ty(t)^{3}\) given \(y(0) = 2\).

  \begin{solution}
    This is a Bernoulli DE, so we need to substitute \(u = y^{1-n}\) where \(n = 3\) here.  Since \(u' = (1-3)y^{-3}y'\), a convenient way to perform the substitution is to multiply the DE through by \((1-3)y^{-3}\). You can write \(1-3\) as \(-2\). I just prefer to do it this way, so I don't forget what I'm doing. Now we have
    \[
      (1-3)y^{-3}y'(t) + t (1-3)y^{-3}y(t) = t(1-3)y^{-3}y(t)^{3}
    \]
    which becomes
    \[
      u'(t) -2t u(t) = -2t.
    \]

    We solve this linear, first-order DE using the integrating factor \(v(t) = e^{\int -2t \;dt} = e^{-t^{2}}\).  Then we have
    \[
      u(t) = e^{t^{2}} \int -2t e^{-t^{2}} \;dt = e^{t^{2}}\left( -e^{-t^{2}} + C \right) = 1 + C e^{-t^{2}}
    \]
    The integration requires a substitution \(w = -t^{2}\).  The initial condition \(u(0) = \left( y(0) \right)^{-2} = 1/4\) implies \(C = -3/4\).  The intermediate solution is 
    \[
      u(t) = 1 - \frac{3}{4}e^{-t^{2}}.
    \]

    Lastly, we recover the original solution using \(y^{-2} = 1 - \frac{3}{4}e^{-t^{2}}\). There are two possible solutions \(y(t) = \pm \left( 1 - \frac{3}{4}e^{-t^{2}} \right)^{-1/2}\). We can discard the negative root since \(y(0)\) is positive. Finally, the solution is 
    \[
      y(t) = \left( 1 - \frac{3}{4}e^{-t^{2}} \right)^{-1/2} 
    \]
  \end{solution}
\end{exercise}

\begin{exercise}
  Solve the initial-value problem \(y'(t) + \frac{2}{t} y(x) = \frac{(y(t))^{4}}{t^{2}}\) and \(y(1) = 1\).

  \begin{solution}
    We have a Bernoulli DE, so substitute \(u = y^{1-4}\) to get
    \[
      u'(t) + \frac{2}{t}(1 - 4) = \frac{1}{t^{2}}(1-4).
    \]
    Simplify a bit to get
    \[
      u'(t) - \frac{6}{t}u(t) = -\frac{3}{t^{2}}.
    \]

    The integrating facto is \(v(t) = e^{\int - \frac{6}{t}\;dt} = t^{-6}\). Then we have
    \[
      u(t) = t^{6} \int - t^{-6} \frac{3}{t^{2}} \;dt = -t^{6} \int 3t^{-8} \;dt = t^{6} \left( \frac{3}{7} t^{-7} + C \right) = \frac{3}{7t} + C t^{6}.
    \]

    Using the initial condition \(u(1) = (y(1))^{-3} = 1\) to find \(C = 4/7\) and \(u(t) = \frac{3}{7t} + \frac{4}{7}t^{6}\).  % You can also write it as \(u(t) = \frac{3 + 4t^{7}}{7t}\).

    Lastly, we recover the original solution
    \[
      y(t) = \left( \frac{3}{7t} + \frac{4}{7} t^{6} \right)^{-1/3}.
    \]
  \end{solution}
\end{exercise}

\begin{exercise}
  Solve the initial-value problem \(y'(t) + \frac{2}{t} y(t) = (t y(t))^{2} \sin(t)\) and \(y(\pi) = 1\).

  \begin{solution}
    Rewrite the Bernoulli DE in the standard form
    \[
      y'(t) + \frac{2}{t} y(t) = t^{2} \left( t(t) \right)^{2}.
    \]

    Substitute \(u = y^{1-2}\) to get
    \[
      u'(t) - \frac{2}{t} u(t) = -t^{2}.
    \]

    The integrating factor is \(v(t) = e^{\int -\frac{2}{t} \;dt} = t^{-2}\). We solve it to get
    \[
      u(t) = t^{2} \int -1 \;dt = t^{2}(-t + C) = -t^{3} + Ct^{2}.
    \]

    Using the initial condition \(u(\pi) = 1\) to get \(C = \frac{\pi + 1}{\pi^{2}}\) and \(u(t) = -t^{3} + \frac{\pi + 1}{\pi^{2}}t^{2}\).

    Recover the original solution to get
    \[
      y(t) = \left( -t^{3} + \frac{\pi + 1}{\pi^{2}}t^{2} \right)^{-1}.
    \]
  \end{solution}
\end{exercise}
