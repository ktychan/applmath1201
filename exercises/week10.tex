\begin{exercise}
  The sample mean and sample standard deviation of last year's final exam were \(70\) and \(12\), respectively.  This year, the mean and standard deviation for the final exam is \(72\) and \(14\). 

  \begin{enumerate}
    \item Which year's students performed better on the final exam.
    \item If A scored \(65\) on this year's final exam, what would be a comparable grade from last year's final exam?
  \end{enumerate}
\end{exercise}

\begin{exercise}
  Suppose we have a set of sample data with standard deviation \(1.37\). If the \(z\)-score of a measurement (value - sample mean) is \(-2.5\).  What is the actual value of the measurement if the sample mean is \(3\)?
\end{exercise}

\begin{exercise}
  Let \(\vec{x}(t)\) be the number of juveniles and adults, respectively, in a population at time \(t\) in years. Every adult in the population produces, on average, \(2\) juveniles in the population next year. Every adult in the population survives to the next year with probability \(1/3\). Every juvenile survives to be an adult next year with probability \(1/2\).

  Develop a discrete-time model for the population.  Find \(\vec{x}(3)\).
\end{exercise}

\begin{exercise}
  Perform the following vector operations pictorially.  Verify that your sketch is correct by performing the operations algebraically.

  \begin{itemize}
    \item \(-3 \begin{bmatrix} 3 \\ 1 \end{bmatrix}\).
    \item \(-2 \begin{bmatrix} 2 \\ -3 \end{bmatrix}\).
    \item \(\begin{bmatrix} -3/2 \\ -4 \end{bmatrix} + \begin{bmatrix} 3 \\ -1/2 \end{bmatrix}\).
    \item \(\begin{bmatrix} 1 \\ -2 \end{bmatrix} - \begin{bmatrix} 2 \\ 3 \end{bmatrix}\).
  \end{itemize}
\end{exercise}

\begin{exercise}
  Find \(a\) that satisfies the matrix-vector equations \(\begin{bmatrix} 1 & a \\ 0 & -1 \end{bmatrix} \begin{bmatrix} 3 \\ 7 \end{bmatrix} = \begin{bmatrix} 38 \\ -7 \end{bmatrix}\).

  % \begin{solution}
  %   Note that \(1 \times 3 + a \times 7 = 38\). Therefore, \(a = 5\).
  % \end{solution}
\end{exercise}

\begin{exercise}
  A company tracks its workforce in three career-stage categories.
  Suppose \(x_{1}(t), x_{2}(t), x_{3}(t)\) represent the number of entry-level, mid-career and senior employees at year \(t\).  
  \begin{itemize}
    \item Senior employees generate \(1.2\) new entry-level hires per senior employee per year.
    \item \(60\%\) of entry-level employees are promoted to mid-career roles each year. The rest leave the company.
    \item \(70\%\) of mid-career employees are promoted to a senior positions each year. \(10\%\) leave the company. The rest stay without promotions.
    \item \(90\%\) of senior employees stays with the business every year.
  \end{itemize}
  Initially, there are \(300\) entry-level, \(200\) mid-career and \(100\) senior employees.  Sketch the transition diagram and model the workforce as a vector recurrence.
\end{exercise}

\begin{exercise}
  A biologist is studying the population dynamics of a species of frogs with three subpopulations tadpoles \(x_{1}(t)\), juveniles \(x_{2}(t)\) and adults \(x_{3}(t)\). Suppose the population dynamics are modelled by the recurrence
  \[
    \vec{x}(t+1) =
    \begin{bmatrix}
      0 & 0 & 120 \\
      0.25 & 0 & 0 \\
      0 & 0.4 & 0.85
    \end{bmatrix}
    \vec{x}(t).
  \]
  Describe, in words, the transition among subpopulations from year \(t\) to year \(t + 1\).
\end{exercise}

\begin{exercise}
  Consider a vector recurrence \(\vec{x}(t+1) = A \vec{x}\).  The arrows in its vector field are described by \(\vec{v} = \begin{bmatrix} 1 & 1 \\ -1 & 1 \end{bmatrix}\vec{x}\) where \(\vec{x}\) represents arbitrary points in the plane. In other words, \(\vec{v}\) is an arrow in the vector field drawn by placing its tail at the tip of \(\vec{x}\). Find \(A\).
\end{exercise}

\begin{exercise}
  Find the unknown constants \(a,b\) in the following equations.

  \begin{itemize}
    \item \(\begin{bmatrix} -13 \\ 18 \end{bmatrix} = a \begin{bmatrix} -2 \\ 5 \end{bmatrix} + b \begin{bmatrix} 3 \\ 7 \end{bmatrix}\).
      % a = 5, b = -1
    \item \(\begin{bmatrix} 1/2 \\ -7/3 \end{bmatrix} = a \begin{bmatrix} 2 \\ -1 \end{bmatrix} + b \begin{bmatrix} -3 \\ 4 \end{bmatrix}\).
      % a = 1/2, b = 2
    \item \(\begin{bmatrix} -5/6 \\ 7/15 \end{bmatrix} = a \begin{bmatrix} 1 \\ 1/5 \end{bmatrix} + b \begin{bmatrix} 1/2 \\ 2 \end{bmatrix}\).
      % a = -1, b = 1/3
    \item \(\begin{bmatrix} 4 \\ 9 \\ -7 \end{bmatrix} = a \begin{bmatrix} 1 \\ -3 \\ 2 \end{bmatrix} + b \begin{bmatrix} -2 \\ 5 \\ 1 \end{bmatrix} + c \begin{bmatrix} 0 \\ 4 \\ -3 \end{bmatrix}\).
  \end{itemize}
\end{exercise}
