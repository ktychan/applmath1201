\begin{exercise}
  Suppose \(A\) is a \(2\)-by-\(2\) matrix and \(\vec{v} = \begin{bmatrix} 2 \\ 1 \end{bmatrix}\) is an eigenvector for \(A\).  Sketch \(\vec{v}\) and \(A\vec{v}\) for each of the scenarios below.
  \begin{enumerate}
    \item The eigenvalue for \(\vec{v}\) is larger than \(1\).
    \item The eigenvalue for \(\vec{v}\) is exactly \(1\).
    \item The eigenvalue for \(\vec{v}\) is between \(0\) and \(1\), exclusive.
    \item The eigenvalue for \(\vec{v}\) is negative.
  \end{enumerate}

  \begin{solution} \phantom{top}

    \begin{enumerate}
      \item The eigenvalue for \(\vec{v}\) is larger than \(1\). Then \(A\vec{v}\) should be longer than \(\vec{v}\).

        \begin{tikzpicture}
          \begin{axis}[basic curve, black, xmin=-2, xmax=6, ymin=-2, ymax=4, xtick distance={2}, ytick distance={2}]
            \addplot coordinates {(0,0)};
            \coordinate (O) at (axis cs:0,0);
            \draw[-Latex, very thick] (O)--(axis cs:2,1) node[below] {\(\vec{v}\)};
            \draw[red, -Latex, very thick] (O)--(axis cs:4,2) node[above] {\(A\vec{v}\)};
          \end{axis}
        \end{tikzpicture} 


      \item The eigenvalue for \(\vec{v}\) is larger than \(1\). Then \(A \vec{v}\) and \(\vec{v}\) overlap.
        \hlwarn{Correction: this case is eigenvalue \emph{exactly} \(1\) (not larger than \(1\)); the sketch \(A\vec{v} = \vec{v}\) is correct.}

        \begin{tikzpicture}
          \begin{axis}[basic curve, black, xmin=-2, xmax=6, ymin=-2, ymax=4, xtick distance={2}, ytick distance={2}]
            \addplot coordinates {(0,0)};
            \coordinate (O) at (axis cs:0,0);
            \draw[-Latex, very thick] (O)--(axis cs:2,1) node[below] {\(\vec{v}\)};
            \draw[red, -Latex, very thick] (O)--(axis cs:2,1) node[above] {\(A\vec{v}\)};
          \end{axis}
        \end{tikzpicture} 

      \item The eigenvalue for \(\vec{v}\) is between \(0\) and \(1\), strict. Then \(A \vec{v}\) should be shorter than \(\vec{v}\) but still point in the same direction.

        \begin{tikzpicture}
          \begin{axis}[basic curve, black, xmin=-2, xmax=6, ymin=-2, ymax=4, xtick distance={2}, ytick distance={2}]
            \addplot coordinates {(0,0)};
            \coordinate (O) at (axis cs:0,0);
            \draw[-Latex, very thick] (O)--(axis cs:2,1) node[below] {\(\vec{v}\)};
            \draw[red, -Latex, very thick] (O)--(axis cs:1,0.5) node[above] {\(A\vec{v}\)};
          \end{axis}
        \end{tikzpicture} 


      \item The eigenvalue for \(\vec{v}\) is negative. Then \(A \vec{v}\) should be point in the opposite direction of \(\vec{v}\).

        \begin{tikzpicture}
          \begin{axis}[basic curve, black, xmin=-2, xmax=6, ymin=-2, ymax=4, xtick distance={2}, ytick distance={2}]
            \addplot coordinates {(0,0)};
            \coordinate (O) at (axis cs:0,0);
            \draw[-Latex, very thick] (O)--(axis cs:2,1) node[below] {\(\vec{v}\)};
            \draw[red, -Latex, very thick] (O)--(axis cs:-1,-0.5) node[below] {\(A\vec{v}\)};
          \end{axis}
        \end{tikzpicture} 
    \end{enumerate}
    
  \end{solution}
\end{exercise}

\begin{exercise}
  Suppose \(\vec{x}(t+1) = A \vec{x}(t)\) models some population dynamics. Suppose further that \(\vec{u}\) and \(\vec{v}\) are two eigenvectors for \(A\). 
  \begin{enumerate}
    \item If \(A \vec{u}\) is longer than \(\vec{u}\), and \(A \vec{v}\) is shorter than \(\vec{v}\), does the population grow unbounded, stabilize, or go extinct in the long run?
    \item If \(A \vec{u}\) is a little bit shorter than \(\vec{u}\), and \(A \vec{v}\) is a lot shorter than \(\vec{v}\), does the population grow unbounded, stabilize, or go extinct in the long run?
  \end{enumerate}

  \begin{solution} The previous exercise should give you an idea how to do this one.

    \begin{enumerate}
      \item Let \(\mu\) be the eigenvalue for \(\vec{u}\) and \(\lambda\) the eigenvalue for \(\vec{v}\).  Then \(\mu > 1\) and \(\mu\) is between \(0\) and \(1\), strict.  Therefore, the dominant eigenvalue is \(\mu\) and is greater than \(1\). The population grows unbounded in the long run.
        \hlwarn{Correction: the second clause should read ``\(\lambda\) is between \(0\) and \(1\)'' (it is \(\lambda\), the eigenvalue of \(\vec{v}\), that is small). The conclusion is unaffected.}
      \item Let \(\mu\) be the eigenvalue for \(\vec{u}\) and \(\lambda\) the eigenvalue for \(\vec{v}\).  Then both \(\mu\)  and \(\lambda\) is between \(0\) and \(1\), strict.  However, \(\mu > \lambda\), and \(\mu\) is the dominant eigenvalue. Therefore, the population goes extinct in the long run.
    \end{enumerate}
  \end{solution}
\end{exercise}

\begin{exercise}
  Let \(x_{1}(t)\) be the number of juveniles in year \(t\), and \(x_{2}(t)\) be the number of adults in year \(t\).  The per-capita birth rate among adults is \(1/4\). The survival rate among adults is \(1/4\). Lastly, \(1/2\) of the juveniles grows into adults in the next year. 

  \begin{enumerate}
    \item Sketch the transition diagram of this population.
    \item Model the population dynamics using a (vector) recurrence.
    \item Solve the recurrence, assuming that \(x_{1}(0) = 32\) and \(x_{2}(0) = 16\).
    \item Determine the long term behaviour of the population. Does the population grow unbounded, stabilize or go extinct? If they don't go extinct, what is the ratio of \(x_{2}(t) / x_{1}(t)\) as \(t\) approaches infinity?
  \end{enumerate}

  \begin{solution} \phantom{top}

    \begin{enumerate}
      \item See below. 

        \begin{tikzpicture}[transition diagram]
          \node[class] (A) at (-1,0) {juveniles\\\(x_{1}\)};
          \node[class] (B) at ( 1,0) {adults\\\(x_{2}\)};

          % forward loop
          \path[->] (A) edge [bend  left] node[above] {\(1/2\)} (B);
          \path[->] (B) edge [bend  left] node[below] {\(1/4\)} (A);
          \path[->] (B) edge [loop right] node[right] {\(1/4\)} (B);  
        \end{tikzpicture}

      \item The model is \(\vec{x}(t+1) = \begin{bmatrix} 0 & 1/4 \\ 1/2 & 1/4 \end{bmatrix} \vec{x}(t)\).
      \item The eigenvectors are \(\vec{u} = \begin{bmatrix} 1 \\2 \end{bmatrix}\) with eigenvalue \(\mu = 1/2\) and \(\vec{v} = \begin{bmatrix} 1 \\ -1 \end{bmatrix}\) with eigenvalue \(-1/4\).

        We solve \(\vec{x}(0) = \begin{bmatrix} 32 \\ 16 \end{bmatrix} = a \vec{u} + b \vec{v}\) to get \(a = 16\) and \(b = 16\). It follows that the solution to the recurrence is
        \[
          \vec{x}(t) = 16 \cdot \left( \frac{1}{2} \right)^{t}\begin{bmatrix} 1 \\2 \end{bmatrix} + 16 \cdot \left( -\frac{1}{4} \right)^{t} \begin{bmatrix} 1 \\ -1 \end{bmatrix}.
        \]
      \item The dominant eigenvalue is \(1/2\) which is smaller than \(1\). Therefore, the population goes extinct in the long-run. 
    \end{enumerate}
  \end{solution}
\end{exercise}

\begin{exercise}
  The traffic lights at a quiet intersection is broken.  At any given time, only the green and the red light up.
  \begin{itemize}
    \item If it's red at time \(t\), then there is a \(20\%\) chance it will turn green at time \(t+1\).
    \item If it's green at time \(t\), then there is a \(50\%\) chance it will turn red at time \(t+1\).
  \end{itemize}

  \begin{enumerate}
    \item Sketch the transition diagram of the traffic light states.
    \item Model the state transitions using a (vector) recurrence.
    \item Solve the recurrence, assuming that \(x_{1}(0) = 0.5\) and \(x_{2}(0) = 0.5\).
    \item Determine \(x_{1}(t)\) and \(x_{2}(t)\) as \(t\) approaches infinity.
  \end{enumerate}

  \begin{solution} \phantom{top}

    \begin{enumerate}
      \item Let \(p_{1}(t)\) be the probability that green lights up at time \(t\), and \(p_{2}(t)\) the probability the red one lights up at time \(t\).

        \begin{tikzpicture}[transition diagram]
          \node[class] (A) at (-1,0) {green\\\(p_{1}\)};
          \node[class] (B) at ( 1,0) {red\\\(p_{2}\)};

          % forward loop
          \path[->] (A) edge [bend  left] node[above] {\(0.5\)} (B);
          \path[->] (A) edge [loop  left] node[left] {\(0.5\)} (A);
          \path[->] (B) edge [bend  left] node[below] {\(0.2\)} (A);
          \path[->] (B) edge [loop right] node[right] {\(0.8\)} (B);  
        \end{tikzpicture}

      \item The model is \(\vec{x}(t+1) = \begin{bmatrix} 0.5 & 0.2 \\ 0.5 & 0.8 \end{bmatrix} \vec{x}(t)\).

      \item The eigenvectors are \(\vec{u} = \begin{bmatrix} 1 \\ 5/2 \end{bmatrix}\) with eigenvalue \(\mu = 1\) and \(\vec{v} = \begin{bmatrix} 1 \\ -1 \end{bmatrix}\) with eigenvalue \(0.3\).

        We solve \(\vec{x}(0) = \begin{bmatrix} 0.5 \\ 0.5 \end{bmatrix} = a \vec{u} + b \vec{v}\) to get \(a = 2/7\) and \(b = 3/14\). It follows that the solution to the recurrence is
        \[
          \vec{x}(t) = \frac{2}{7} \cdot \begin{bmatrix} 1 \\ 5/2 \end{bmatrix} + \frac{3}{14} \cdot \left( \frac{3}{10} \right)^{t} \begin{bmatrix} 1 \\ -1 \end{bmatrix}.
        \]

      \item The dominant eigenvalue is \(1\). Note \(\lim_{t \to \infty} \left( \frac{3}{10} \right)^{t} = 0\). Therefore, \(\vec{x}(t)\) approach \(\begin{bmatrix} 2/7 \\ 5/7 \end{bmatrix}\) as \(t\) approach infinity.
    \end{enumerate}
  \end{solution}
\end{exercise}

\begin{exercise}
  Predict the long-term behaviour of the model \(\vec{x}(t+1) = \begin{bmatrix} 2 & -3/2 \\ -12 & -1\end{bmatrix} \vec{x}(t)\) with \(x_{1}(0) = 3\) and \(x_{2}(0) = 6\). 

  \begin{solution}
    The eigenvalues of the matrix are \(\mu = 5\) and \(\lambda = -4\). Therefore, \(\mu\) is the dominant eigenvalue, and both components of \(\vec{x}(t)\) approaches infinity as \(t\) approaches infinity.

    \hlwarn{Correction: the dominant eigenvector is \([1, -2]^{\top}\), so its components have opposite signs: \(x_{1}(t) \to +\infty\) while \(x_{2}(t) \to -\infty\). (Both grow without bound in magnitude.)}
  \end{solution}
\end{exercise}

\begin{exercise}
  Consider the recurrence \(\vec{x}(t+1) = \begin{bmatrix} 5 & 4 \\ 1 & 2\end{bmatrix} \vec{x}(t)\). If \(\mu = 1\) and \(\lambda = 6\) are its eigenvalues, find the ratio \(x_{2}(t) / x_{1}(t)\) as \(t\) approaches infinity.

  \begin{solution}
    The dominant eigenvalue \(\mu = 6\) has an eigenvector \(\begin{bmatrix} 1 \\ 1/4 \end{bmatrix}\). Therefore, both components of \(\vec{x}(t)\) approach infinity, and \(x_{2}(t)/x_{1}(t)\) approaches \(1/4\) as \(t\) tends to infinity.

    \hlwarn{Correction: the dominant eigenvalue is \(\lambda = 6\) (the problem labels it \(\lambda\), not \(\mu\)); the eigenvector and the ratio \(1/4\) are correct.}
  \end{solution}
\end{exercise}

\begin{exercise}
  Solve the initial-value problem \( \frac{d\vec{x}}{dt} = \begin{bmatrix} 3 & 0  \\ 0 & -2 \end{bmatrix} \vec{x}(t) \) with \(\vec{x}(0) = \begin{bmatrix} 1/2 \\ 1/3 \end{bmatrix}\).

  \begin{solution}
    The eigenvalues are \(\mu = 3\) and \(\lambda = 2\) with eigenvectors \(\vec{v} = \begin{bmatrix} 1 \\ 0 \end{bmatrix}\) and \(\begin{bmatrix} 0 \\ 1 \end{bmatrix}\) respectively.  The initial condition \(\vec{x}(0)\) can be written in terms of the eigenvectors as follows
    \[
      \vec{x}(0) = \frac{1}{2} \vec{u} + \frac{1}{3} \vec{v}.
    \]
    Therefore, the solution is
    \[
      \vec{x}(t) = \frac{1}{2} \cdot e^{3t} \begin{bmatrix} 1 \\ 0 \end{bmatrix} + \frac{1}{3} \cdot e^{2t} \begin{bmatrix} 0 \\ 1 \end{bmatrix}.
    \]

    \hlwarn{Correction: the second eigenvalue is \(-2\), not \(2\), so the second term should be \(\frac{1}{3} e^{-2t} \begin{bmatrix} 0 \\ 1 \end{bmatrix}\).}
  \end{solution}
\end{exercise}

\begin{exercise}
  Find the general solutions for
  \(
  \begin{array}{rcrcr}
    x'_{1}(t) &=& -2 x_{1}(t) &+&  x_{2}(t) \\
    x'_{2}(t) &=&    x_{1}(t) &+& 2x_{2}(t)
  \end{array}
  \).

  \begin{solution}
    The system of differential equations is \(\vec{x}'(t) = \begin{bmatrix} -2 & 1 \\ 1 & 2 \end{bmatrix} \vec{x}(t)\). The matrix has eigenvectors and eigenvalues
    \[
      \vec{u} = \begin{bmatrix}
        -2 + \sqrt{5} \\
        1
      \end{bmatrix}
      \text{ with \(\mu = \sqrt{5}\)}
      \quad\text{and}\quad
      \vec{v} = \begin{bmatrix}
        -2 - \sqrt{5} \\
        1
      \end{bmatrix}
      \text{ with \(\lambda = -\sqrt{5}\)}.
    \]
    Therefore, the general solution is
    \[
      \vec{x}(t) = 
      \alpha \cdot e^{\sqrt{5}t}
      \begin{bmatrix}
        -2 + \sqrt{5} \\
        1
      \end{bmatrix}
      + 
      \beta \alpha \cdot e^{-\sqrt{5}t}
      \begin{bmatrix}
        -2 - \sqrt{5} \\
        1
      \end{bmatrix},
    \]
    where \(\alpha,\beta\) are constants.

    \hlwarn{Correction: the coefficient on the second term should be \(\beta\), not \(\beta\alpha\).}
  \end{solution}
\end{exercise}

\begin{exercise}
  Find the specific solutions for
  \(
  \begin{array}{rcrcr}
    u'_{1}(t) &=& 2 u_{1}(t) &+& 2u_{2}(t) \\
    u'_{2}(t) &=&   u_{1}(t) &+& 3u_{2}(t)
  \end{array}
  \) with \(
  \begin{array}{rcr}
    u_{1}(0) &=&  1 \\
    u_{2}(0) &=&  3 \\
  \end{array}
  \).

  \begin{solution}
    The system of differential equations is \(\vec{u}'(t) = \begin{bmatrix} 2 & 2 \\ 1 & 3 \end{bmatrix} \vec{u}(t)\). The matrix has eigenvectors and eigenvalues
    \[
      \begin{bmatrix}
        1 \\
        1
      \end{bmatrix}
      \text{ with \(\mu = 4\)}
      \quad\text{and}\quad
      \begin{bmatrix}
        -2 \\
        1
      \end{bmatrix}
      \text{ with \(\lambda = 1\)}.
    \]

    Solve for \(\alpha,\beta\) in \(\vec{u}(0) = \begin{bmatrix} 1 \\ 1 \end{bmatrix} + \beta \begin{bmatrix} -2 \\ 1 \end{bmatrix}\) to get \(\alpha = 7/3\) and \(\beta = 2/3\).  Therefore, the solution is
    \hlwarn{Correction: the equation should read \(\vec{u}(0) = \begin{bmatrix} 1 \\ 3 \end{bmatrix} = \alpha\begin{bmatrix} 1 \\ 1 \end{bmatrix} + \beta\begin{bmatrix} -2 \\ 1 \end{bmatrix}\) (the left side is \([1,3]^{\top}\) and the first vector needs the coefficient \(\alpha\)); the values \(\alpha=7/3, \beta=2/3\) are correct.}
    \[
      \vec{u}(t) = 
      \frac{7}{3} \cdot e^{4t} 
      \begin{bmatrix}
        1 \\
        1
      \end{bmatrix}
      +
      \frac{2}{3} \cdot e^{t} 
      \begin{bmatrix}
        -2 \\
        1
      \end{bmatrix}.
    \]

  \end{solution}
\end{exercise}
