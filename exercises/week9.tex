\begin{exercise}
  Suppose \(E\) and \(F\) are events in a sample space and \(\mathbb{P}(E) = 0.4\) and \(\mathbb{F} = 0.6\). Which statements are true?

  \begin{enumerate}
    \item \(E\) and \(F\) are mutually exclusive.
    \item \(E\) and \(F\) are not mutually exclusive.
    \item There is not enough information to determine if \(E\) and \(F\) are mutually exclusive.
    \item \(E\) and \(F\) are independent.
    \item \(E\) and \(F\) are not independent.
    \item There is not enough information to determine if \(E\) and \(F\) are independent.
  \end{enumerate}

  \begin{solution}
    There is not enough information to determine if \(E\) and \(F\) are mutually exclusive or independent.
  \end{solution}
\end{exercise}

\begin{exercise}
  Consider the following probability density function for a continuous random variable.
  \[
    f(x) =
    \begin{cases}
      \dfrac{2}{9} (x - 2), & 2 \le x \le 5, \\[6pt]
      0, & \text{otherwise}.
    \end{cases}
  \]

  \begin{enumerate}
    \item Find its cumulative distribution function.
    \item Calculate its expectation, variance and standard deviation.
  \end{enumerate}

  \begin{solution}
    Recall that the CDF is defined by \(F(x) = \mathbb{P}(X \le x)\).

    \begin{enumerate}
      \item Note that by definition \(\mathbb{P}(X \le x) = \int_{-\infty}^{x} f(x) \;dx\). Therefore, we calculate
        \[
          F(x) = \int_{2}^{x} \frac{2}{9}(t - 2) \;dt = \left[ \frac{2}{9}\left( \frac{x^{2}}{2} - 2x \right) \right]_{2}^{x} = \frac{1}{9}(x - 2)^{2}.
        \]

      \item Recall the formulas, and that we only have to integrate over wherever the function is not zero.

        For the expectation, we have
        \[
          \mu = E(X) = \int_{-\infty}^{\infty} xf(x) \;dx = \int_{2}^{5} \frac{2}{9}(x - 2)x \;dx = 4.
        \]

        For the variance, we have
        \[
          V(x) = \int_{-\infty}^{\infty} (x - \mu)^{2} f(x) \;dx = \int_{2}^{5} (x - 4)^{2} \frac{2}{9}(x - 2) \;dx = \frac{1}{2}.
        \]
        
        Lastly, for the standard deviation, we have
        \[
          \sigma = \sqrt{V(X)} = \frac{1}{\sqrt{2}}.
        \]
    \end{enumerate}
  \end{solution}
\end{exercise}

\begin{exercise}
  The probability density function for a continuous random variable \(X\) is given below.
  \[
    f(x) =
    \begin{cases}
      \dfrac{1}{2} x, & 0 \le x \le 1, \\[8pt]
      \dfrac{1}{2}(2x - 4), & 2 \le x \le 3, \\[8pt]
      \dfrac{1}{2}(6 - x), & 5 \le x \le 6, \\[8pt]
      0, & \text{otherwise}.
    \end{cases}
  \]

  \begin{enumerate}
    \item Find its cumulative distribution function.
    \item Calculate \(\mathbb{P}(1/2 \le X < 5.5)\).
    \item Calculate its expectation, variance and standard deviation.
  \end{enumerate}

  \begin{solution}
    This is all about the definitions of PDF and CDF.

    \begin{itemize}
      \item By definition \(F(x) = \mathbb{P}(X \le x) = \int_{-\infty}^{x} f(x) \;dx\).  
        \begin{itemize}
          \item If \(0 \le x \le 1\), then we have
            \[
              F(x) = \int_{0}^{x} \frac{1}{2}t \;dt = \frac{x^{2}}{4}.
            \]

          \item If \(2 \le x \le 3\), then we have
            \[
              F(x) = F(1) + \int_{2}^{x} \frac{1}{2}(2t - 4) \;dt = \frac{1}{4} + \frac{1}{2}(x - 2)^{2}.
            \]

          \item If \(5 \le x \le 6\), then we have
            \[
              F(x) = F(3) + \int_{5}^{x} \frac{1}{2}(6 - t)\;dt = \frac{1}{4} + \frac{1}{2} - \frac{x^{2}}{4} + 3x - 35/4.
            \]
        \end{itemize}

        Therefore, we have
        \[
          F(x) = 
          \begin{cases}
            0 & \text{if } x < 0 \\[1ex]
            \frac{x^{2}}{4} & \text{if } 0 \le x \le 1, \\[1ex]
            \frac{1}{4} & \text{if } 1 < x \le 2, \\[1ex]
            \frac{1}{4} + \frac{1}{2}(x - 2)^{2} & \text{if } 2 \le x \le 3, \\[1ex]
            \frac{1}{4} + \frac{1}{2} & \text{if } \text{if } 3 < x \le 5, \\[1ex]
            \frac{1}{4} + \frac{1}{2} - \frac{x^{2}}{4} + 3x - 35/4 & \text{if } 5 \le x \le 6 \\[1ex]
            1 &\text{ if } x > 6
          \end{cases}
        \]


        \begin{tikzpicture}
          \begin{axis}[grid=both, xmin=-1, xmax=6.5, ymin=-0.1, ymax=1.1, title={PDF}, minor tick num=1]
            \addplot[main, very thick, domain=0:1] {1/2*x};
            \addplot[supp, very thick, domain=2:3] {1/2*(2*x - 4)};
            \addplot[warn, very thick, domain=5:6] {1/2*(6 - x)};
          \end{axis}
        \end{tikzpicture}
        \begin{tikzpicture}
          \begin{axis}[grid=both, xmin=-1, xmax=6.5, ymin=-0.1, ymax=1.1, title={CDF}, minor tick num=1]
            \addplot[very thick, black, domain=-1:0] {0};
            \addplot[main, very thick, domain=0:1] {x^2/4};
            \addplot[very thick, black, domain=1:2] {1/4};
            \addplot[supp, very thick, domain=2:3] {1/4 + 1/2*(x-2)^2};
            \addplot[very thick, black, domain=3:5] {1/4 + 1/2};
            \addplot[warn, very thick, domain=5:6] {1/4 + 1/2 - x^2/4 + 3*x - 35/4};
            \addplot[very thick, black, domain=6:7] {1};
          \end{axis}
        \end{tikzpicture}

      \item This calculation is a bit tedious. since we have to split the integrals over three separate intervals. 

        For the expectation, we have
        \begin{align*}
          & \mu = E(X) \\
          =& \int_{-\infty}^{\infty} xf(x) \;dx &\text{(the definition)}\\
          =& \int_{0}^{1} x \frac{1}{2}x \;dx + \int_{2}^{3} x\frac{1}{2}(2x - 4) \;dx + \int_{5}^{6} x \frac{1}{2}(6 - x) &\text{(split)}\\
          =& \frac{1}{6} + \frac{4}{3} + \frac{4}{3} &\text{(integrate)} \\
          =& \frac{17}{6}.
        \end{align*}

        For the variance, we have
        \begin{align*}
          & V(X)  \\
          = & \int_{-\infty}^{\infty} (x - \mu)^{2}f(x) \;dx &\text{(the definition)}\\
          = & \int_{0}^{1} \left(x - \frac{17}{6}\right)^{2} \frac{1}{2}x \;dx + \int_{2}^{3} \left(x - \frac{17}{6}\right)^{2} \frac{1}{2}(2x - 4) \;dx + \int_{5}^{6} \left(x - \frac{17}{6}\right)^{2} \frac{1}{2}(6 - x) \;dx &\text{(split)}\\
          = & \frac{19}{16} + \frac{1}{24} + \frac{227}{144} &\text{(integrate)} \\
          = & \frac{101}{36}.
        \end{align*}

        For the standard deviation, we have
        \begin{align*}
          \sigma = \sqrt{\frac{101}{36}}.
        \end{align*}
    \end{itemize}
  \end{solution}
\end{exercise}

\begin{exercise}
  The probability density function of a continuous random variable is given below. Find the unknown constant \(a\).

  \[
    f(x) =
    \begin{cases}
      a x, & 0 \le x \le 2, \\[8pt]
      0.35\, (6 - x), & 4 \le x \le 6, \\[8pt]
      0, & \text{otherwise}.
    \end{cases}
  \]

  \begin{solution}
    A valid PDF must satisfy \(\int_{-\infty}^{\infty} f(x) \;dx = 1\). Therefore,  we solve for \(a\) in the equation
    \[
      \int_{0}^{2} ax \;dx + \int_{4}^{6} 0.35 (6 - x) \;dx = 1.
    \]
    Complete the integration to get \(2a + 0.7 = 1\). Therefore, \(a = 0.15\).
  \end{solution}
\end{exercise}

  \begin{solution}

\begin{exercise}

    \item \(f(x)\) is always increasing.

\begin{exercise}
  A game is played by flipping a fair coin three times. Each time heads is flipped, you receive one cookie. Each time tails is flipped you receive \(5\) cookies. You play this game once.

  Model the game using a random variable.

  \begin{enumerate}
    \item List all outcomes of the underlying sample space of \(X\).
    \item Describe the probability mass function for \(X\).
    \item Find \(E(X)\).
  \end{enumerate}

  \begin{solution}
    Let \(X\) be the random variable.

    \begin{enumerate}
      \item The sample space is \(\{HHH, HHT, HTH, HTT, THH, THT, TTH, TTT\}\). These are all possible outcomes of flipping a coin three times.
      \item Enumerating all outcomes tells us the payouts.

        \begin{tabular}{c|l}
          outcome & payout \\\midrule
          HHH & \(3\) \\\midrule
          HHT & \(7\) \\\midrule
          HTH & \(7\) \\\midrule
          HTT & \(11\) \\\midrule
          THH & \(7\)  \\\midrule
          THT & \(11\) \\\midrule
          TTH & \(11\) \\\midrule
          TTT & \(15\) 
        \end{tabular}

        Therefore, the PMF is
        \[
          f(x) = 
          \begin{cases}
            1/8 & \text{if } x = 3 \\
            3/8 & \text{if } x = 7 \\
            3/8 & \text{if } x = 11 \\
            1/8 & \text{if } x = 15
          \end{cases}
        \]

      \item The expectation is \(E(X) = 3 \cdot \frac{1}{8} + 7 \cdot \frac{3}{8} + 11 \cdot \frac{3}{8} + 15 \cdot \frac{1}{8} = \frac{72}{8} = 9\).
    \end{enumerate}

  \end{solution}
\end{exercise}
