\begin{document} \section{Differential equations, linear and first-order}

Solving a differential equation is one way to make sense of the unknown function. Basic calculus and a neat trick tell us how to solve linear, first-order differential equations \hlwarn{Edited: ``equation'' \(\to\) ``equations''}.

\begin{thm} \label{thm:diff-eq-linear-first-order}
  We wish to solve for the unknown function \(u(t)\) in
  \begin{equation} \label{eq:diff-eq-linear-first-order}
    u'(t) + p(t) u(t) = q(t).
  \end{equation}
  
  An \hlmain{integrating factor} is
  \begin{equation} \label{eq:diff-eq-integrating-factor}
    v(t) = e^{\int p(t) \;dt} \text{ and choose \(C = 0\) for convenience}.
  \end{equation} 

  The general solution to Equation~\eqref{eq:diff-eq-linear-first-order} is
  \begin{equation} \label{eq:diff-eq-linear-first-order-solution1}
    u(t) = \frac{1}{v(t)} \left( \int v(t) q(t) \;dt + C \right).
  \end{equation}
\end{thm}

\begin{example}
  Solve the IVP \(G'(t) = 1 - 0.1 G(t)\) and \(G(0) = 0\) from Example~\ref{eg:diff-eq-model-intro}.

  \blanklines{20}
\end{example}

{\footnotesize
Interpret the ``\(+C\)'' in Equation~\eqref{eq:diff-eq-linear-first-order-solution1} according to your preference.

\begin{enumerate}
  \item You expect yourself to forget to add a ``\(+ C\)'' after evaluating an indefinite integral.

    In such a case, remember Equation~\eqref{eq:diff-eq-linear-first-order-solution1} as stated above and omit the integration constant after evaluating \(\int v(t) q(t) \;dt\). Leaving the ``\(+C\)'' in the equation serves as a reminder that an integration constant should be there.

  \item You expect yourself to always include a ``\(+C\)'' after evaluating an indefinite integral.

    In such a case, remember Equation~\eqref{eq:diff-eq-linear-first-order-solution1} as \(u(t) = \frac{1}{v(t)}\left( \int v(t) q(t) \;dt \right)\) because you don't need an extra reminder that an indefinite integral represents infinitely many antiderivatives of the integrand. You should also remember that \(\textstyle \int 0 \;dt = C\).
\end{enumerate}
}


\clearpage

Theorem~\ref{thm:diff-eq-linear-first-order} is the compact form of Method~\ref{method:integrating-factor}.

\begin{method} \label{method:integrating-factor}
  \hlmain{An integrating factor} for a linear, first-order differential equation \(u'(t) + p(t) u(t) = q(t)\) is
  \begin{equation} 
    \color{main}
    v(t) = e^{\int p(t) \;dt} \text{ and choose \(C = 0\) for convenience}. \tag{\ref{eq:diff-eq-integrating-factor}}
  \end{equation}
  To find the general solution for the differential equation, multiply the given linear, first-order DE through by the integrating factor \(v(t)\) and reorganize the left-hand side as the result of the product rule. The \emph{general solution} of the differential equation, Equation~\eqref{eq:diff-eq-linear-first-order-solution1}, is the result of integrating both sides then isolating for \(u(t)\).
\end{method}
{\footnotesize \faExclamationTriangle{} Theorem~\ref{thm:diff-eq-linear-first-order} and Method~\ref{method:integrating-factor} only apply to linear, first-order DEs in the form \(u'(t) + p(t) u(t) = q(t)\).}

\begin{example} \label{ex:diff-eq-linear-first-order-1}
  Find the general solution for \( u'(t) + \frac{2}{t} u(t) = t^{2}\).
  
  \begin{enumerate}[wide, label=(Step~\arabic*)]
    \item Identify an integrating factor \(v(t)\).

      \blanklines{5}

    \item Apply Method~\ref{method:integrating-factor} or Equation~\eqref{eq:diff-eq-linear-first-order-solution1}.

      \blanklines{30}
  \end{enumerate}
  \vfill{}
\end{example}
