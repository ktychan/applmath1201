\begin{document} \section{Discrete random variable}

A random variable is discrete if all possible values of \(X(\text{outcome})\) form a discrete set. Discrete random variables \hlwarn{Edited: ``Discrete random variable are'' \(\to\) ``Discrete random variables are''} are described by their PMF or CDF which are two different but commonly used ways to represent the same underlying information in a discrete random variable.

\begin{definition}[PMF and CDF]
  Let \(X\) be a discrete random variable.

  The \hlmain{probability mass function} (PMF) for \(X\) is a nonnegative function \(f_{X}(x)\) so that
  \[
    \mathbb{P}({\color{main} X = x})
    \quad\text{is}\quad
    f_{X}(x).
  \]

  The \hlmain{cumulative distribution function} (CDF) for \(X\) is a nonnegative and non-decreasing function \(F_{X}(x)\) so that
  \[
    \mathbb{P}({\color{main} X \le x})
    \quad\text{is}\quad
    F_{X}(x).
  \]
\end{definition}

We can think of random variables as aggregate \hlwarn{Edited: ``aggregates forms'' \(\to\) ``aggregate forms''} forms of data.  Outcomes in the sample space are grouped by common associated data (e.g., height).  \hlmain{Expectation}, \hlmain{variance} and \hlmain{standard deviation} \hlwarn{Edited: ``standard deviations'' \(\to\) ``standard deviation''} are common ways to describe the shape of data.

\begin{definition}
  Suppose \(X\) is a discrete random variable with values \(x_{1}, \ldots, x_{n}\) and a PMF \(f_{X}(x)\).
  \begin{enumerate}[itemsep={1ex}]
    \item Its expectation is \(\mu = E(X) = \sum_{i=1}^{n} x_{i} f_{X}(x_{i})\).
    \item Its variance is \(V(X) = E\left( (X - \mu)^{2} \right) = \sum_{i=1}^{n} (x_{i} - \mu)^{2} f_{X}(x_{i})\).
    \item Its standard deviation is \(\sigma = \sqrt{V(X)}\). \hlwarn{Edited: ``\(V(x)\)'' \(\to\) ``\(V(X)\)''}
  \end{enumerate}

  Variance is often denoted as \(\sigma^{2}\).
\end{definition}

\textbf{Summary} of discrete random variable \(X\) whose values are sorted as \(x_{1} < \cdots < x_{n}\).
\begin{itemize}
  \item PMF \(f_{X}(x)\) is \(\mathbb{P}(X = x)\).
  \item CDF \(F_{X}(x)\) is \(\mathbb{P}(X \le x)\).
  \item Relations between PMF and CDF.
    \begin{itemize}
      \item \(F_{X}(x_{i}) = \underbrace{f_{X}(x_{1}) + \cdots + f_{X}(x_{i-1})}_{F_{X}(x_{i-1})} + f_{X}(x_{i}) = F_{X}(x_{i-1}) + f_{X}(x_{i})\).
      \item \(f_{X}(x_{i}) = F_{X}(x_{i}) - F_{X}(x_{i-1})\).
    \end{itemize}
  \item Expectation \(\mu = E(X) = \sum_{i=1}^{n} x_{i} f_{X}(x_{i})\).
  \item Variance \(V(X) = E\left( (X - \mu)^{2} \right) = \sum_{i=1}^{n} (x_{i} - \mu)^{2} f_{X}(x_{i})\). Variance is often denoted as \(\sigma^{2}\).
  \item Standard deviation \(\sigma = \sqrt{V(X)}\).
\end{itemize}
